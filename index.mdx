import { useState, useEffect } from 'react';

export const ButtonGroup = ({options = [], selected, onClick, fillWidth = false, type = 'default', ...props}) => {
    return (
        <div
            className={`
        inline-flex flex-row items-center justify-center gap-1
        ${type === 'default' ? 'border border-white/10 bg-[#f5f5f5] dark:bg-[#1a1a1a] p-1' : 'p-0'}
        ${fillWidth ? 'w-full' : 'w-auto'}
        rounded-lg relative z-20
      `}
            {...props}
        >
            {options.map(({ value, label, ...buttonProps }) => (
                <button
                    key={value}
                    onClick={() => onClick && onClick(value)}
                    aria-pressed={value === selected}
                    className={`
            flex items-center justify-center
            px-4 py-2 rounded-md
            font-medium text-sm
            border-none cursor-pointer
            transition-all duration-200
            ${fillWidth ? 'flex-1' : ''}
            ${
                        value === selected
                            ? 'bg-[#e5e5e5] text-gray-900 dark:bg-[#3a3a3a] dark:text-white'
                            : 'bg-transparent text-gray-400 hover:bg-[#f0f0f0] hover:text-gray-900 dark:hover:bg-[#2a2a2a] dark:hover:text-white'
                    }
          `}
                    {...buttonProps}
                >
                    {label}
                </button>
            ))}
        </div>
    );
}

export const HeroCard = ({ filename, title, description, href, links, icon }) => {
    const [isDark, setIsDark] = useState(false);

    useEffect(() => {
        if (typeof window !== 'undefined') {
            const checkDark = () => {
                setIsDark(document.documentElement.classList.contains('dark'));
            };
            checkDark();
            const observer = new MutationObserver(checkDark);
            observer.observe(document.documentElement, { attributes: true, attributeFilter: ['class'] });
            return () => observer.disconnect();
        }
    }, []);

    const darkFilename = filename ? filename.replace(/(\.[^.]+)$/, "--dark$1") : null;

    const cardStyle = {
        padding: '32px 16px 16px 16px',
        borderRadius: '8px',
        backgroundColor: isDark ? '#1b1b18' : '#f9f9f9',
        border: isDark ? '1px solid #3c3c3c' : '1px solid #e5e7eb',
        display: 'block',
        textDecoration: 'none',
        color: 'inherit'
    };

    const content = (
        <>
            {icon ? (
                <div style={{ marginBottom: '24px' }}>
                    {icon}
                </div>
            ) : (
                <>
                    <img
                        src={`https://ch-docs-product-selector.s3.eu-central-1.amazonaws.com/${filename}`}
                        style={{
                            display: isDark ? 'none' : 'block',
                            width: '100%',
                            height: '192px',
                            objectFit: 'cover',
                            borderRadius: '8px',
                            border: '1px solid #d1d5db',
                            pointerEvents: 'none'
                        }}
                    />
                    <img
                        src={`https://ch-docs-product-selector.s3.eu-central-1.amazonaws.com/${darkFilename}`}
                        style={{
                            display: isDark ? 'block' : 'none',
                            width: '100%',
                            height: '192px',
                            objectFit: 'cover',
                            borderRadius: '8px',
                            border: '1px solid #494949',
                            pointerEvents: 'none'
                        }}
                    />
                </>
            )}
            <h3 style={{
                marginTop: icon ? '0' : '20px',
                color: isDark ? '#FFFFFF' : '#111827',
                fontSize: '24px',
                fontWeight: '600',
                lineHeight: '1.3'
            }}>
                {title}
            </h3>
            <span style={{
                display: 'block',
                marginTop: '16px',
                fontSize: '14px',
                fontWeight: '400',
                color: isDark ? '#FFFFFF' : '#4b5563',
                minHeight: '3rem'
            }}>{description}</span>
            {links && links.length > 0 && (
                <ul style={{ marginTop: '24px', listStyle: 'none', padding: 0 }}>
                    {links.map((link, index) => (
                        <li key={index} style={{ marginTop: index > 0 ? '12px' : '0' }}>
                            <a href={link.href} style={{
                                display: 'flex',
                                alignItems: 'center',
                                justifyContent: 'space-between',
                                color: isDark ? '#FCFF74' : '#374151',
                                fontSize: '14px',
                                fontWeight: '500',
                                textDecoration: 'none'
                            }}>
                                <span>{link.label}</span>
                                <svg xmlns="http://www.w3.org/2000/svg" width="16" height="16" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round" style={{ flexShrink: 0 }}>
                                    <path d="M5 12h14M12 5l7 7-7 7"/>
                                </svg>
                            </a>
                        </li>
                    ))}
                </ul>
            )}
        </>
    );

    if (links && links.length > 0) {
        return <div style={cardStyle}>{content}</div>;
    }

    return <a style={cardStyle} href={href}>{content}</a>;
};
